#include <Servo.h>

#define trigPin 2
#define echoPin 3

long duration;
int distance;

Servo myservo;

int calculateDistance() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);
  duration = pulseIn(echoPin, HIGH);
  distance = duration * 0.034 / 2;
  return distance;
}

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  myservo.attach(4);
  Serial.begin(9600);
}

void loop() {
  // Fast sweep from 0째 to 180째
  for (int pos = 0; pos <= 180; pos += 5) {
    myservo.write(pos);
    delay(5);                // faster motion
    calculateDistance();
    Serial.print(pos);
    Serial.print(",");
    Serial.print(distance);
    Serial.print(".");
  }

  // Fast sweep from 180째 back to 0째
  for (int pos = 180; pos >= 0; pos -= 5) {
    myservo.write(pos);
    delay(5);
    calculateDistance();
    Serial.print(pos);
    Serial.print(",");
    Serial.print(distance);
    Serial.print(".");
  }
}
